// LOAD ALL VIDEOS async function loadFeed() { const res = await fetch("/videos"); const videos = await res.json(); const container = document.createElement("div"); videos.forEach(v => { const vid = document.createElement("video"); vid.src = v; vid.width = 300; vid.controls = true; container.appendChild(vid); }); document.body.appendChild(container); } // run on load window.onload = loadFeed;
